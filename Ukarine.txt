ll fun(ll idx,string &S,map<string,char>&M)
{
    if(idx>=sz(S))
        return 1;

    string s1=S.substr(idx,1);

    ll ans=0;

    if(M.find(s1)!=M.end())
    {
        ans+=fun(idx+1,S,M);
    }

    if(idx+1<sz(S))
    {
        string s2=S.substr(idx,2);

        if(M.find(s2)!=M.end())
            ans+=fun(idx+2,S,M);
    }

    return ans;
}
